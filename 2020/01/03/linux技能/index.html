1. 请执行命令取出linux中eth0的IP地址

方法1：ifconfig eth0| grep 'inet addr'| awk -F":" '{print $2}'|awk '{print $1}'
方法2：ifconfig eth0| grep 'inet addr'| cut -d: -f2|cut -d" " -f1
方法3：ifconfig eth0| grep 'inet addr'| cut -d: -f2| awk '{print $1}'
方法4：ifconfig eth0| awk -F '[ :]+' 'NR==2 {print $4}'

2. 统计apache日志单ip访问请求数排名

方法1：awk '{S[$1]++} END {for(a in S) print S[a],a}' access.log |sort -rn -k2
方法2：awk '{print $1}' access.log |sort |uniq -c |sort -rn
方法3: sed 's/- -.*$//g' access.log |sort |uniq -c |sort -rn

3. 处理以下文件内容,将域名取出并根据域名进行计数排序处理(百度和sohu面试题)

[root@ywxi test]# cat oldboy.log 
http://www.etiantian.org/index.html
http://www.etiantian.org/1.html
http://post.etiantian.org/index.html
http://mp3.etiantian.org/index.html
http://www.etiantian.org/3.html
http://post.etiantian.org/2.html
方法1： cut -d "/" -f3 test.log |sort |uniq -c    
方法2： awk -F "/" '{print $3}' test.log |sort |uniq -c
方法3： sed 's#^http://##g' test.log|sed 's#/.*##g'|sort |uniq -c
方法4： cat test.log |tr "\/" "\n"|grep etiantian|sort|uniq -c
方法5： sed -e 's#^.*//\(.*etiantian.*\)/.*html#\1#g' test.log|sort|uniq -c|sort
方法6： cut -d "/" -f 3 test.log |awk '{S[$1]++} END {for(a in S) print S[a],a}'
方法7： awk -F "/" '{++S[$3]} END {for(key in S) print S[key],key}' test.log
方法8： awk -F '[:/]+' '{print $2}' test.log |sort |uniq -c

4. 找出当前目录下包含127.0.0.1关键字的文件

方法1: grep -rl '127.0.0.1' ./
方法2: awk '/127.0.0.1/{print FILENAME}' `find -type f`
方法3: find -type f |xargs grep “127.0.0.1”
方法4: find -type f -exec grep “127.0.0.1” {} \;

5. 删除jfedu.txt行前和行尾的任意空格

sed ‘s/^[ \t]*//；s/[ \t]*$//’ jfedu.txt

6. 将root和bash动作替换

sed -ne ‘s/root/admin/’ -ne ‘s/bash/sh/p’ /etc/passwd

7. 在每行开头插入#号 ，直接修改源文件

sed -i ‘s/^/#/’ /etc/passwd

8. 将第1~5行内容迁移至末尾

sed ‘1,5(H:d);$G’ test.txt

9. 显示/etc/passwd文件中不以/bin/bash结尾的行

cat /etc/passwd |grep -v /bin/bash$

10. 找出/etc/passwd中的两位或三位数

grep -o "\<[1-9][0-9]\{1,2\}\>" /etc/passwd|sort -nr|uniq

11. 查找root和sshd的行

grep -e ‘root’ -e ’sshd’ /etc/passwd

12. 过滤文件中空白行与#开头的行

egrep -v ‘^$|^#’ /etc/passwd

13. 显示三个用户root、 mage、 wang的UID和默认shell

egrep "^((root)|(mage)|(wang)\>)" /etc/passwd

14. 查文件大小大于1Mb的文件

find /data/ -size +1M

15. 查文件类型并将权限设置为644

find /data/ -type f|xargs chmod 644 -R

16. 查找/data/目录以大写字母开头的文件

find /data/ -name “[A-Z]*”

17. 在/home下查最近两天内改动过的文件

find /home  -mtime   -2

18. 查找在系统中属于FRED这个用户的文件

find /   -user    fred

19. 输出以冒号为分隔的/etc/passwd文件中记录的字段段数

awk -F: '{print $0,NF}' /etc/passwd

20. 输出以冒号为分隔符，以root开头的行第一列，且在前面插入“Hi,”

awk -F: '/^root/{print "Hi," $1}' /etc/passwd
